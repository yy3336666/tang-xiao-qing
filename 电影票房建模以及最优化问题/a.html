def kendalltau_movies(df):
    corr1=np.zeros((df.shape[1],df.shape[1]))
    for i in range(corr1.shape[0]):
        for j in range(corr1.shape[1]):
            corr1[i,j]=kendalltau(df.iloc[:,i],df.iloc[:,j])[0]
    return corr1
